// 프로그래머스 고득점 합격 Kit : 정렬_가장 큰 수
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

bool cmp(string x, string y) {
	string c1, c2;
	c1 = x + y;
	c2 = y + x;
	return c1 > c2;
}

string solution(vector<int> numbers) {
	string answer = "";

	// int형을 숫자를 string으로 바꾸어 s_numbers벡터에 삽입
	vector<string> s_numbers;
	for (int tmp : numbers) {
		s_numbers.push_back(to_string(tmp));
	}

	// cmp함수 조건에 따라 정렬
	sort(s_numbers.begin(), s_numbers.end(), cmp);
	for (string number : s_numbers) {
		answer += number;
	}

	// 입력된 모든 숫자가 0이였을 경우 ("0000..."이 답으로 안나오도록 함)
	if (answer[0] == '0') answer = "0";

	return answer;
}
