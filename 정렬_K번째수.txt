#include <string>
#include <vector>
#include <algorithm>

using namespace std;

vector<int> solution(vector<int> array, vector<vector<int>> commands) {
	vector<int> answer; // 출력답안
	
	for (int i = 0; i < commands.size(); i++) {
		vector<int> slice; // 주어진 배열의 일부를 저장
		for (int j = commands[i][0] -1; j < commands[i][1]; j++) {
			slice.push_back(array[j]);
		}
		sort(slice.begin(), slice.end()); // 오름차순 정렬
		answer.push_back(slice[commands[i][2]-1]); // k번째 수를 넣음 (K : 1~slice.size())
	}

	return answer;
}
