// 프로그래머스 고득점 합격 Kit : 탐욕법_큰 수 만들기
#include <string>
#include <vector>

using namespace std;

string solution(string number, int k) {
	int notFixing = 0;
	while (k != 0) {
		int max = number[notFixing];
		int maxIndex = notFixing;
		for (int i = notFixing; i <= notFixing+k; i++) {
			if (number[i] > max) {
				max = number[i];
				maxIndex = i;
			}
		}
		number = number.substr(0, notFixing) + number.substr(maxIndex);
		k -= (maxIndex - notFixing);
		notFixing++;

		// 테스트 12번 케이스 처리
		if (notFixing == number.size() - k && k != 0) {
			number = number.substr(0, number.size() - k);
			break;
		}
	}

	string answer = number;
	return answer;
}
