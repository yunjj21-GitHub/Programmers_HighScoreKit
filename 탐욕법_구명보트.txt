#include <string>
#include <vector>
#include <algorithm>
#include <deque>

using namespace std;

bool cmp(int x, int y) {
	return x > y;
}

int solution(vector<int> people, int limit) {
	int answer = 0;

	deque<int> rescue; // 구조해야 하는 사람들의 몸무게를 저장
	for (int person : people) {
		rescue.push_front(person);
	}
	sort(rescue.begin(), rescue.end(), cmp); // 오름차순 정렬

	while (!rescue.empty()) { 
		if (rescue.front() + rescue.back() <= limit) {
			rescue.pop_front();
			rescue.pop_back();
		}
		else {
			rescue.pop_front();
		}
		answer++;

		if (rescue.size() == 1) { // 구출해야 하는 사람이 1명밖에 남지 않았다면
			rescue.pop_front();
			answer++;
		}
	}

	return answer;
}
