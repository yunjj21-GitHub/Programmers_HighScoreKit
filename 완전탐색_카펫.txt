#include <string>
#include <vector>
#include <cmath>

using namespace std;

vector<int> solution(int brown, int yellow) {
	vector<int> answer;

	/*
	* 조건1, x와 y는 자연수이다.
	* 조건2, x는 y보다 크거나 같다.
	* 조건3, x*y = 갈색 격자의 수 + 노란색 격자의 수
	* 조건4, x+y = (갈색 격자의 수 + 4) / 2
	*/

	int xy = brown + yellow;
	int x_sum_y = (brown + 4) / 2;
	int x_start = ceil((float)x_sum_y / 2);
	for (int x = x_start; x < x_sum_y; x++) {
		int y = x_sum_y - x;
		if (x*y == xy) {
			answer.push_back(x);
			answer.push_back(y);
		}
	}

	return answer;
}
