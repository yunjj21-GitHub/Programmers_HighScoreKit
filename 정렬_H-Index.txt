#include <string>
#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;

bool cmp(int x, int y) {
	return x > y;
}

int solution(vector<int> citations) {
	int answer = 0;

	// 인용된 횟수를 기준으로 내림차순 정렬
	sort(citations.begin(), citations.end(), cmp);

	// h번 이상 인용된 논문의 개수가 h개 만큼은 있어야 h의 후보가 될 수 있다.
	// 따라서 h의 범위 : 0~n
	int n = citations.size();
	for (int h = n; h >= 0; h--) {
		int index = 0;
		while (index < n && citations[index] >= h) {
			index++;
		}
		if (index >= h) {
			answer = h;
			break;
		}
	}

	return answer;
}
